170  inline int uncaught_exception_count()
171  {
172 #if defined(SCOPE_HAS_UNCAUGHT_EXCEPTIONS)
173  // C++17 implementation
174  return std::uncaught_exceptions();
175 #elif defined(SCOPE_HAS_CXA_GET_GLOBALS)
176  // Tested on {clang 3.2,GCC 3.5.6,GCC 4.1.2,GCC 4.4.6,GCC 4.4.7}x{x32,x64}
177  using byte = unsigned char;
178  auto count = int{};
179  const auto* ptr = reinterpret_cast<const byte*>(::abi::__cxa_get_globals()) + sizeof(void*);
180 
181  // __cxa_eh_globals::uncaughtExceptions, x32 offset - 0x4, x64 - 0x8
182  std::memcpy(&count, ptr, sizeof(count));
183  return count;
184 #elif defined(SCOPE_HAS_GETPTD)
185  // MSVC specific. Tested on {MSVC2005SP1,MSVC2008SP1,MSVC2010SP1,MSVC2012}x{x32,x64}.
186  using byte = unsigned char;
187  auto count = int{};
188 
189  const auto offset = (sizeof(void*) == 8u ? 0x100 : 0x90);
190  const auto* ptr = static_cast<const byte*>(::scope::detail::_getptd()) + offset;
191 
192  // _tiddata::_ProcessingThrow, x32 offset - 0x90, x64 - 0x100
193  std::memcpy(&count, ptr, sizeof(count));
194  return count;
195 #else
196  // Portable C++03 implementation. Does not allow to detect multiple
197  // nested exceptions.
198 
199  // This is a buggy fallback since it will only work with 1 exception
200  // in-flight, but we don't have any other options without exploiting
201  // internal compiler features.
202  return static_cast<int>(std::uncaught_exception());
203 #endif
